# Set the name of the OU to add the group to
$ou = "OU=Users-Prop,OU=Aspen,DC=aspen,DC=com"

# Set the name of the group to add to users
$groupName = "AzPwChg"

# Get all the user accounts in the OU
$users = Get-ADUser -Filter * -SearchBase $ou

# Add the group to each user
foreach ($user in $users) {
    Add-ADGroupMember -Identity "$groupName" -Members $user
}

# Output a message to confirm the script has completed
Write-Host "The $groupName group has been added to all users in $ou."
