


# Import the Active Directory module
Import-Module ActiveDirectory

# Specify the target Organizational Unit (OU)
$ou = "OU=TestingforAlex,OU=MIS,OU=Users-HO,OU=Aspen,DC=aspen,DC=com"

# Get the user accounts in the specified OU
$users = Get-ADUser -Filter * -SearchBase $ou -Properties Department

# Loop through the user accounts
foreach ($user in $users) {
    $username = $user.SamAccountName
    $department = $user.Department

    # Clear the description field
    $user | Set-ADUser -Clear Description

    # Update the description field with the department
    $user | Set-ADUser -Description $department

    # Output the result
    Write-Output "Updated description field for user: $username"
}
